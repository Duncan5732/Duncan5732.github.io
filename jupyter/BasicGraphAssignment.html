data = df_pres['State'].values

#Finding the number of times presidents are born in specific states
freq = {}
for i in data:
    if i in freq:
        freq[i] += 1
    else:
        freq[i] = 1
        
#Sorting the aforementioned dictionary by the frequency of states (descending order)      
sort_freq = sorted(freq.items(), key=lambda x: x[1], reverse=True)
 
#This number could be adjusted and symbolizes the number of states appearing on the graph           
numPlot = 4
sort_freq = sort_freq[:numPlot]

#Setting the x and y variables for graphing purposes
x = [k for [k,j] in sort_freq]          # use this variable for the x-axis data
y = [j for [k,j] in sort_freq]          # use this variable for the y-axis data
xs = range(len(sort_freq))

# Creation of variables for the figure and axes
fig, ax = plt.subplots()

#Adjustments to the axes of the graph
ax.bar(xs, y, align='center', color='blue')
ax.xaxis.set_label_text("U.S. States", fontsize=25)
ax.yaxis.set_label_text("Count", fontsize=25)
ax.set_xticks(xs)
ax.set_xticklabels(x, rotation = 45)    #Putting the x-axis tickmarks at an angle
ax.xaxis.set_tick_params(labelsize=12)
ax.yaxis.set_tick_params(labelsize=12)
ax.spines['right'].set_visible(False)   #Removes the exterior lines on the right and top
ax.spines['top'].set_visible(False)

#Adjustments to the figure of the graph
fig.suptitle("Most Popular Birth States of U.S. Presidents", fontsize=20)
fig.set_size_inches(9, 5)

#Displays the bar graph
plt.show()

# This frequency histogram displays the four most frequent states that U.S. presidents are born in (24 of the 44 -- 54.55%)
